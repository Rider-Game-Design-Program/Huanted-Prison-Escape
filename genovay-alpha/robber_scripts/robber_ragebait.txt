var random_number = RandomNumberGenerator.new()

func enter():
	super.enter()
	owner.set_physics_process(true)
	robber_sprite.play("Ragebait")
	await robber_sprite.animation_finished
	object_spawn_position = object_spawnpoint.position
	print("Spawn Position: ", object_spawn_position)
	Global.inventory_max = Global.player_inventory_amount
	#print("Ragebait State Entered")
	ragebait()
	can_transition = false

func exit():
	super.exit()
	owner.set_physics_process(false)
	#print("Ragebait State Exited")

func ragebait():
	
	instance_lighter = lighter_scene.instantiate()
	tablet_instance = tablet_scene.instantiate()
	headphones_instance = headphones_scene.instantiate()
	controller_instance = controller_scene.instantiate()
	plushie_instance = plushie_scene.instantiate()
	shotgun_instance = shotgun_scene.instantiate()
	diary_instance = diary_scene.instantiate()
	blueprint_instance = blueprint_scene.instantiate()
	photo_instance = photo_scene.instantiate()
	
	#print("Current inventory Slot 1: ", resource.inventorySlots[0].inventory_item.itemName)
	
	random_number.randomize()
	var slot_picked = (random_number.randi_range(Global.inventory_min, Global.inventory_max))
	#print(random_number.randi_range(Global.inventory_min, Global.inventory_max))
	#print("Slot Picked: ", slot_picked)
	
	if resource.inventorySlots[slot_picked].inventory_item.itemName == "Lighter":
		#print("Lighter")
		instance_lighter.position = object_spawn_position
		add_child(instance_lighter)
	elif resource.inventorySlots[slot_picked].inventory_item.itemName == "Drawing Tablet":
		#print("Tablet")
		tablet_instance.position = object_spawn_position
		add_child(tablet_instance)
	elif resource.inventorySlots[slot_picked].inventory_item.itemName == "Headphones":
		#print("Headphones")
		headphones_instance.position = object_spawn_position
		add_child(headphones_instance)
	elif resource.inventorySlots[slot_picked].inventory_item.itemName == "Controller":
		#print("Controller")
		controller_instance.position = object_spawn_position
		add_child(controller_instance)
	elif resource.inventorySlots[slot_picked].inventory_item.itemName == "Plushie":
		#print("Plushie")
		plushie_instance.position = object_spawn_position
		add_child(plushie_instance)
	elif resource.inventorySlots[slot_picked].inventory_item.itemName == "Player Shotgun":
		#print("Plushie")
		shotgun_instance.position = object_spawn_position
		add_child(shotgun_instance)
	elif resource.inventorySlots[slot_picked].inventory_item.itemName == "Diary":
		#print("Plushie")
		diary_instance.position = object_spawn_position
		add_child(diary_instance)
	elif resource.inventorySlots[slot_picked].inventory_item.itemName == "Blueprint":
		#print("Plushie")
		blueprint_instance.position = object_spawn_position
		add_child(blueprint_instance)
	elif resource.inventorySlots[slot_picked].inventory_item.itemName == "Photo":
		#print("Plushie")
		photo_instance.position = object_spawn_position
		add_child(photo_instance)
	else:
		print("Null")

func transition():
	#print("Robber Dazed: ", Global.robber_dazed)
	if can_transition == true:
		can_transition = false
		Global.robber_dazed = false
		#Global.robber_still = false
		get_parent().change_state("Movement")

func _on_animated_sprite_2d_animation_finished() -> void:
	can_transition = true
	Global.robber_still = false
	robber_sprite.play("Prep")
	await get_tree().create_timer(2.0).timeout
	transition()
	#get_parent().change_state("Movement")
