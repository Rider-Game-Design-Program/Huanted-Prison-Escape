extends AudioStreamPlayer

@export var music_node: AudioStreamPlayer

const level_soundtrack = preload("res://lemaire-milestone-01/Capstone Main Level 1.wav")
const warden_soundtrack = preload("res://lemaire-milestone-01/Boss_Fight_Soundtracks/Capstone.Warden.Fight.wav")
const robber_soundtrack = preload("res://lemaire-milestone-01/Boss_Fight_Soundtracks/Capstone Robber Fight (extended).wav")
const arsonist_soundtrack = preload("res://lemaire-milestone-01/Boss_Fight_Soundtracks/Capstone Arsonist Fight.wav")

# Plays music track.
func play_music(Music: AudioStream, volume = 0.0):
	if stream == Music:
		pass
	volume_db = volume
	play()

# Selects which track to play.
func play_level_music():
	if !Global.play_warden_sfx and !Global.play_robber_sfx and !Global.play_arsonist_sfx:
		music_node.stream = level_soundtrack
		play_music(level_soundtrack, -8.0)
	
	# Plays when player triggers Warden Boss Battle.
	elif Global.play_warden_sfx:
		music_node.stream = warden_soundtrack
		play_music(warden_soundtrack, -6.0)
	
	# Plays when player triggers Robber Boss Battle.
	elif Global.play_robber_sfx:
		music_node.stream = robber_soundtrack
		play_music(warden_soundtrack, -18.0)
	
	elif Global.play_arsonist_sfx:
		music_node.stream = arsonist_soundtrack
		play_music(arsonist_soundtrack, -18.0)
