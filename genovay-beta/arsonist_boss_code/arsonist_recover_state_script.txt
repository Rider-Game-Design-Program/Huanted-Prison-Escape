extends State
class_name arsonist_recover

func enter():
	#print("STATE CHANGE")
	Global.arsonist_can_move = false
	await get_tree().create_timer(2.8).timeout
	Global.arsonist_can_move = true
	transition()

func exit():
	super.exit()
	owner.set_physics_process(false)

func transition():
	if Global.arsonist_can_move == true:
		print("RECOVER")
		get_parent().change_state("Move_Decider")
