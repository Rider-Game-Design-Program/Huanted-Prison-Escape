extends CharacterBody2D
class_name Basic_Enemy

@export var enemy_characterbody2d: CharacterBody2D
@export var player_characterbody: CharacterBody2D
@export var player_health_scene: CharacterBody2D
@export var player_punch_area: Area2D

@onready var punch_area = $Punch_Detection_Area
@onready var enemy_punch_attack_area = $AttackArea
@onready var enemy_progressbar = $ProgressBar
@onready var death_sfx = $Sfx/Death_Sfx
@onready var punch_sfx = $Sfx/Punch_Sfx

const SPEED = 30
var health = 100:
	set(value):
		health = value
		enemy_progressbar.value = value
#var current_health: int

var current_state = IDLE
var direction = Vector2.RIGHT
var start_position
var is_roaming = true
var player
var player_entered: bool = false
var player_engaged: bool = false
var dir: Vector2
var enemy_direction
var enemy_dead: bool = false

enum {
	IDLE, 
	NEW_DIRECTION,
	MOVE
}

# Runs at start.
func _ready() -> void:
	randomize()
	start_position = position # position = position of NPC
	enemy_progressbar.visible = false

# Runs every frame.
func _process(delta: float) -> void:
	enemy_direction = enemy_characterbody2d.position.x
	
	if health <= 0:
		pass
	else:
		if current_state == 0 or current_state == 1 and is_roaming: # 0 = IDLE, 1= NEW_DIRECTION
			$AnimatedSprite2D.play("Idle")
		if current_state == 2 and is_roaming:
			if direction.x == -1:
				$AnimatedSprite2D.flip_h = false
				$AnimatedSprite2D.play("Walk")
				enemy_punch_attack_area.scale.x = 1
			if direction.x == 1:
				$AnimatedSprite2D.play("Walk")
				$AnimatedSprite2D.flip_h = true
				enemy_punch_attack_area.scale.x = -1
	
		if is_roaming:
			match current_state:
				IDLE:
					pass
				NEW_DIRECTION:
					direction = choose([Vector2.RIGHT, Vector2.LEFT]) # Can add Vector2.UP and Vector2.DOWN if needed.
				MOVE:
					move(delta)
					
		if !is_roaming and player_entered:
			if !player_engaged:
				$AnimatedSprite2D.play("Walk")
				velocity = position.direction_to(player_characterbody.position) * SPEED
				dir.x = abs(velocity.x) / velocity.x
				move_and_slide()
			
				if enemy_direction > player_characterbody.position.x:
					$AnimatedSprite2D.flip_h = false
					
				else:
					$AnimatedSprite2D.flip_h = true
				
			else:
				$AnimatedSprite2D.play("Punch")
				punch_sfx.play()

# Creates an array and shuffles it, return the first value after shuffling.
func choose(array):
	array.shuffle() # Mixes array around.
	return array.front() # Returns first value from the shuffle.

func move(delta):
	position += direction * SPEED * delta

func _on_timer_timeout() -> void:
	$Timer.wait_time = choose([0.5, 1.0, 1.5])
	current_state = choose([IDLE, NEW_DIRECTION, MOVE])

func _on_player_detection_body_entered(body: Node2D) -> void:
	enemy_progressbar.visible = true
	is_roaming = false
	player_entered = true
	
func _on_player_detection_body_exited(body: Node2D) -> void:
	enemy_progressbar.visible = false
	is_roaming = true
	player_entered = false

func _on_punch_area_body_entered(body: Node2D) -> void:
	player_engaged = true

func _on_punch_area_body_exited(body: Node2D) -> void:
	player_engaged = false

func toggle_damage_collisions():
	var damage_zone_collision = enemy_punch_attack_area.get_node("PunchCollision") #"CollisionShape2D"
	var wait_time: float
	
	wait_time = 0.65 # 6 divided by 18 because the animation is 5 frames at 18fps.
	damage_zone_collision.disabled = false
	await get_tree().create_timer(wait_time).timeout # creates a timer.
	damage_zone_collision.disabled = true

func _on_attack_area_body_entered(body: Node2D) -> void:
	player_health_scene.apply_damage(10)

func _on_animated_sprite_2d_animation_finished() -> void:
	if player_engaged:
		toggle_damage_collisions()
	if enemy_dead:
		queue_free()

func _on_hitbox_area_body_entered(body: Node2D) -> void:
	player_health_scene.apply_damage(10)

func _on_hitbox_area_area_entered(area: Area2D) -> void:
	health -= 34
	
	if health <= 0:
		enemy_dead = true
		$AnimatedSprite2D.play("Death")
		await get_tree().create_timer(0.9).timeout
		death_sfx.play()
