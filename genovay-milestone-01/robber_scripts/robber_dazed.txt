extends State

@export var robber_sprite: AnimatedSprite2D
@export var robber_hitbox: CollisionShape2D

@onready var dazed_sfx = $Dazed_Sfx

func enter(): 
	super.enter()
	owner.set_physics_process(true)
	robber_sprite.play("Dazed")
	
	# Shrinks the hitbox area during dazed only, currently stops player from
	# recieving damage upon hitbox contact, may want to change and create a 
	# deal damage area so we aren't shrinking the enemy hitbox.
	robber_hitbox.scale.x = 0.5
	robber_hitbox.scale.y = 0.5
	
	dazed_sfx.play()

func exit():
	super.exit()
	owner.set_physics_process(false)

func transition():
	if Global.robber_still == false:
		robber_hitbox.scale.x = 1.0
		robber_hitbox.scale.y = 1.0
		get_parent().change_state("Ragebait")
		#print("Transition. Robber: Dazed: ", Global.robber_dazed)
