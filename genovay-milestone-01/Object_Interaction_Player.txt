extends Control

signal object_interaction_finished

@export_file("*.json") var object_file

var object = []
var current_object_dialogue_id = 0
var object_active = false

func _ready() -> void:
	$NinePatchRect.visible = false

func start():
	if object:
		return
	object_active = true
	$NinePatchRect.visible = true
	object = load_dialogue()
	current_object_dialogue_id = -1
	next_script()

func _input(event: InputEvent) -> void:
	if !object_active:
		return
	if event.is_action_pressed("ui_accept"): # Use enter key.
		next_script()

func next_script():
	current_object_dialogue_id += 1
	if current_object_dialogue_id >= len(object):
		object_active = false
		$NinePatchRect.visible = false
		emit_signal("object_interaction_finished")
		return
	
	$NinePatchRect/ItemName.text = object[current_object_dialogue_id]['item name']
	$NinePatchRect/Text.text = object[current_object_dialogue_id]['text']
